# Day 32 - DSA Problem Solutions

## Question 1 ( In Github )  
**Problem:**  
Implement push and pop operations on a stack and verify stack operations.

**Code:**  
```c
#include <stdio.h>

#define MAX 100

int stack[MAX];
int top = -1;

// Push operation
void push(int value) {
    if (top == MAX - 1) {
        printf("Stack Overflow\n");
    } else {
        top++;
        stack[top] = value;
    }
}

// Pop operation
void pop() {
    if (top == -1) {
        printf("Stack Underflow\n");
    } else {
        top--;
    }
}

// Display operation
void display() {
    if (top == -1) {
        printf("Stack is Empty\n");
    } else {
        for (int i = top; i >= 0; i--) {
            printf("%d ", stack[i]);
        }
        printf("\n");
    }
}

int main() {
    int n;
    scanf("%d", &n);

    while(n--){
        int val;
        scanf("%d",&val);
        push(val);

    }
    int m;
    scanf("%d",&m);
    while(m--){
        pop();
    }
    display();

    return 0;
}
```
