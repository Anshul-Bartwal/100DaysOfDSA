# Day 38 - DSA Problem Solutions

## Question 1 ( In Github )  
**Problem:**  
Deque (Double-Ended Queue)

**Code:**  
```c
#include <stdio.h>

#define MAX 100

int dq[MAX];
int front=-1,rear=-1;

// Check full
int isFull() {
    return(front==(rear + 1)%MAX);
}

// Check empty
int isEmpty() {
    return(front==-1);
}

// Insert at fronk
void insertFront(int x) {
    if (isFull()) {
        printf("Deque Overflow\n");
        return;
    }

    if (isEmpty()) {
        front = rear = 0;
    } else {
        front = (front - 1 + MAX) % MAX;
    }

    dq[front] = x;
}

// Insert at rear
void insertRear(int x) {
    if(isFull()) {
        printf("Deque Overflow\n");
        return;
    }

    if(isEmpty()) {
        front = rear = 0;
    } else{
        rear=(rear + 1)% MAX;
    }

    dq[rear] = x;
}

// Delete from front
void deleteFront() {
    if(isEmpty()) {
        printf("Deque Underflow\n");
        return;
    }

    if(front == rear) {
        front = rear = -1;
    }else{
        front = (front + 1) % MAX;
    }
}

// Delete from rear
void deleteRear() {
    if (isEmpty()) {
        printf("Deque Underflow\n");
        return;
    }

    if (front==rear) {
        front = rear = -1;
    } else {
        rear = (rear - 1 + MAX) % MAX;
    }
}

// Display
void display() {
    if (isEmpty()) {
        printf("Deque is Empty\n");
        return;
    }

    int i = front;
    while (1) {
        printf("%d ", dq[i]);
        if (i == rear) break;
        i = (i + 1) % MAX;
    }
    printf("\n");
}

int main() {
    insertRear(10);
    insertRear(20);
    insertFront(5);
    insertRear(30);

    display();  // 5 10 20 30

    deleteFront();
    display();  // 10 20 30

    deleteRear();
    display();  // 10 20

    return 0;
}
```

**Sample Output:**  
```
5 10 20 30 
10 20 30 
10 20 
```
