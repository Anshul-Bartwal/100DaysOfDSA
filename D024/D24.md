## Question 1 ( In Github )  
**Problem:**  
First Occurrence of a Key - Implement using linked list with dynamic memory allocation.

**Code:**  
```c
#include <stdio.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node * next;
};
void insertEnd(struct Node* head,int val){
    if (head==NULL){
        printf("Really???");
        return;
    }
    struct Node* temp=head;
    struct Node* newNode=NULL;
    newNode=(struct Node*)malloc(sizeof(struct Node));
    while(temp->next!=NULL){
        temp=temp->next;
    }
    temp->next=newNode;
    newNode->data=val;
    newNode->next=NULL;
}
void printList(struct Node* head){
    if (head==NULL || head->next==NULL){
        printf("Really???");
        return;
    }
    struct Node* temp=head->next;
    while(temp->next!=NULL){
        printf("%d ",temp->data);
        temp=temp->next;
    }
    if(temp==head){
        printf("really?");
        return;
    }
    printf("%d",temp->data);

}
void freeList(struct Node* head){
    if (head==NULL){
        printf("Really???");
        return;
    }
    struct Node* temp=head;
    while(head!=NULL){
        temp=head;
        head=head->next;
        free(temp);

    }
}
int removeKey(struct Node*head,int key){
    if(head==NULL || head->next==NULL){
        printf("Really?");
        return 0;
    }
    struct Node* temp=head;
    while(temp->next!=NULL){
        if(temp->next->data==key){
            struct Node* found=temp->next;


            
            temp->next=temp->next->next;
            
            free(found);
            return 1;

        }
        temp=temp->next;
    }
}
int main(){
    struct Node* head;
    head=(struct Node*)malloc(sizeof(struct Node));
    head->next=NULL;
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        int val;
        scanf("%d",&val);
        insertEnd(head,val);
    }

    int key;
    scanf("%d",&key);
    removeKey(head,key);
    printList(head);
    freeList(head);
    return 0;
}
```
