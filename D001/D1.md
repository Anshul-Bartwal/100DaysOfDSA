## Question 1 ( In Github )  
**Problem:**  
Write a C program to insert an element x at a given 1-based position pos in an array of n integers. Shift existing elements to the right to make space.

**Code:**  
```c
#include <stdio.h>
#include <stdlib.h>
int main(){
    int n;
    printf("Enter n: ");
    scanf("%d",&n);
    int *arr = (int *)calloc(n,sizeof(int));
    printf("Enter array elements: ");
    for(int k=0;k<n;k++){
        scanf("%d",&arr[k]);
    }
    int i,val;
    printf("Enter i and value");
    scanf("%d",&i);
    scanf("%d",&val);
    int *temp = realloc(arr,(n+1)*sizeof(int));
    arr=temp;
    for(int k=n-1;k>=i;k--){
        arr[k+1]=arr[k];
    }


    
    arr[i] = val;

    for(int k=0;k<n;k++){
        printf("%d",arr[k]);
    }
    free(arr);
    return 0;
    

}
```
