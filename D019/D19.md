## Question 1 ( In Github )  
**Problem:**  
Given an array of integers, find two elements whose sum is closest to zero.

**Concept Used**
- Two-pointer technique

**Code:**  
```c
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>
int compare(const void*a,const void* b){
    return (*(int *)a- *(int *)b);
}

int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for (int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    qsort(arr,n,sizeof(int),compare);

    int left=0,right=n-1;
    int resL,resR;
    int minSum=INT_MAX;

    while(left<right){
        int sum=arr[left]+arr[right];
        if(abs(sum) < minSum){
            minSum=abs(sum);
            resL=arr[left];
            resR=arr[right];
        }
        if(sum>0){
            right--;
        }else if(sum<0){
            left++;
        }else{

            
            resL=arr[left];
            resR=arr[right];
            break;
        }
    
    }
    printf("%d %d\n", resL, resR);
    return 0;
}
```

**Sample Output:**  
```
4
1 2 -3 -2
-2 2

```
