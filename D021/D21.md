## Question 1 ( In Github )  
**Problem:**  
Create and Traverse Singly Linked List

**Concept Used**
- Linked List

**Code:**  
```c
#include <stdio.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node * next;
};
void insertEnd(struct Node* head,int val){
    if (head==NULL){
        printf("Really???");
        return;
    }
    struct Node* temp=head;
    struct Node* newNode=NULL;
    newNode=(struct Node*)malloc(sizeof(struct Node));
    while(temp->next!=NULL){
        temp=temp->next;
    }
    temp->next=newNode;
    newNode->data=val;
    newNode->next=NULL;
}
void printList(struct Node* head){
    if (head==NULL){
        printf("Really???");
        return;
    }
    struct Node* temp=head;
    while(temp->next!=NULL){
        printf("%d ",temp->next->data);
        temp=temp->next;
    }

    // printf("%d",temp->data);

}
void freeList(struct Node* head){
    if (head==NULL){
        printf("Really???");
        return;
    }
    struct Node* temp=head;
    while(head!=NULL){
        temp=head;
        head=head->next;
        free(temp);

    }
}
int main(){
    struct Node* head=NULL;
    head=(struct Node*)malloc(sizeof(struct Node));
    head->next=NULL;
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        int val;
        scanf("%d",&val);
        insertEnd(head,val);
    }
    printList(head);
    freeList(head);
    return 0;


}

```
